using System;
using System.Collections.Generic;

class Employee
{                                                         // class (Structure)
    public string Name;
    public int Id;
    public double Salary;

    public virtual void CalculateSalary()                  //  Polymorphism (Method overrriding) - base class
    {
        Console.WriteLine("Base Salary: " + Salary);
    }

    public void Display()                                  //  Encapsualtion -   class employee combines data + function in one unit 
    {
        Console.WriteLine("Employee ID: " + Id);
        Console.WriteLine("Employee Name: " + Name);   
    }
}

class Manager : Employee                                      // Hierarchical inheritance (two child class inherits parent class)
{
    public override void CalculateSalary()                    // Abstraction - calculation logic hides inside the method 
    {
        double bonus = Salary * 0.20;
        Console.WriteLine("Manager Salary with Bonus: " + (Salary + bonus));
    }
}

class Developer : Employee
{
    public override void CalculateSalary()
    {
        double incentive = Salary * 0.10;
        Console.WriteLine("Developer Salary with Incentive: " + (Salary + incentive));
    }
}

class Program
{
    static void Main()
    {
        List<Employee> employees = new List<Employee>();           // Generic collecion class to store multiple values of same data type dynamically 

        Console.Write("Enter number of employees: ");
        int n = Convert.ToInt32(Console.ReadLine());

        for (int i = 0; i < n; i++)
        {
            Console.WriteLine("\nSelect Employee Type");
            Console.WriteLine("1. Manager");
            Console.WriteLine("2. Developer");
            Console.WriteLine();

            Console.WriteLine("Option :");
            int choice = Convert.ToInt32(Console.ReadLine());
            Console.WriteLine();

            Employee emp;                                             //object creation  - instance of a class 

            if (choice == 1)
                emp = new Manager();
            else
                emp = new Developer();

            Console.Write("Enter Employee ID: ");
            emp.Id = Convert.ToInt32(Console.ReadLine());

            Console.Write("Enter Employee Name: ");
            emp.Name = Console.ReadLine();

            Console.Write("Enter Base Salary: ");
            emp.Salary = Convert.ToDouble(Console.ReadLine());

            employees.Add(emp);
        }

        Console.WriteLine("\n------ All Employee Details ------");

        foreach (Employee emp in employees)
        {
            emp.Display();
            emp.CalculateSalary();
            Console.WriteLine("----------------------");
        }
    }
}
