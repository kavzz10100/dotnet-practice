[HttpGet]
public async Task<IActionResult> GetAll()
{
    return Ok(await _service.GetAllEmployees());
}

[HttpGet("{id}")]
public async Task<IActionResult> GetById(int id)
{
    return Ok(await _service.GetEmployeeById(id));
}

[HttpPost]
public async Task<IActionResult> CreateEmployee(EmployeeDto dto)
{
    var result = await _service.CreateEmployee(dto);
    return Ok(result);
}

[HttpPut("{id}")]
public async Task<IActionResult> Update(int id, Employee emp)
{
    return Ok(await _service.UpdateEmployee(id, emp));
}

[HttpDelete("{id}")]
public async Task<IActionResult> Delete(int id)
{
    return Ok(await _service.DeleteEmployee(id));
}
